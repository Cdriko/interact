mousePressed() and mousePressed test

int value = 0;

void setup(){
  size(200, 200);
}

void draw() {
  fill(value);
  rect(125, 125, 50, 50);
  
  if (mousePressed == true) {
    fill(0);
  } else {
    fill(255);
  }
  rect(25, 25, 50, 50);
}

void mousePressed() {
  if(value == 0) {
    value = 255;
  } else {
    value = 0;
  }
}
