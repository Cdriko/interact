void drawNextObject()
{
  if(isDone){
    isDone = false;
    objCounter = 1;
    x = startX;
    y = startY;
  }

  x += 10;
  
  if( objCounter % NUM_OBJECTS_PER_ROW-1 == 0 )
  {
    y += 10;
    x = startX;
  }
  
  pushMatrix();
  translate(x, y, 320);
  sphere(5);
  popMatrix();

  objCounter++;
}
