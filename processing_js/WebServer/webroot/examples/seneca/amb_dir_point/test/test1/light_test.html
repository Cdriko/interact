sphere() & pointLight() Test

This test was taken from Processing.org which tests to make
sure the pointLight() works correctly on a sphere. 

// Test from Processing.org
void setup() {
  size(100, 100, P3D);
}

void draw() {
  background(0);
  noStroke();
  pointLight(51, 102, 126, 35, 40, 36);
  noStroke();
  translate(80, 50, 0);
  sphere(30);
}
